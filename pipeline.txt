pipeline {
    agent any

    stages {

        stage('PULL') {
            steps {
                git 'https://github.com/rohanmeshram50-png/new-repo.git'
                echo 'Pull Success'
            }
        }

        stage('BUILD') {
            steps {
                dir('backend') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('TEST') {
            steps {
                dir('backend') {
                    withSonarQubeEnv('SonarQube') {
                        sh '''
                            mvn sonar:sonar \
                            -Dsonar.projectKey=Student-app \
                            -Dsonar.projectName=Student-app \
                        '''
                    }
                }
                echo 'Test + Sonar Success'
            }
        }

        stage('DEPLOY') {
            steps {
                echo 'Deploy Success'
            }
        }
    }
}
